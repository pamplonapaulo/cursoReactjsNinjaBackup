'use strict'
 
const filterRecursive = (arr = [], func = (item) => item) => {
 
  return (function internal(arrayInternal, counter) {
    const [head, ...tail] = arrayInternal
 
    return arrayInternal.length == 0 ? [] : [
      func(head, counter, arr),
      ...internal(tail, counter + 1)
    ]
  })(arr, 0)
}
 
export default filterRecursive
